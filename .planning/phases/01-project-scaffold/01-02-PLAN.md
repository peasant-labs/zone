   ```yaml
   version: 2

   builds:
     - id: zone
       main: .
       binary: zone
       env:
         - CGO_ENABLED=0
       goos:
         - linux
         - darwin
         - windows
       goarch:
         - amd64
         - arm64
       ignore:
         - goos: windows
           goarch: arm64
       ldflags:
         - -s -w
         - -X main.version={{.Version}}
         - -X main.commit={{.Commit}}
         - -X main.date={{.Date}}

   archives:
     - id: zone
       name_template: "{{ .ProjectName }}_{{ .Os }}_{{ .Arch }}"
       format_overrides:
         - goos: windows
           formats: [zip]

   checksum:
     name_template: "checksums.txt"

   homebrew_casks:
     - name: zone
       ids:
         - zone
       repository:
         owner: peasant-labs
         name: homebrew-tap
       homepage: "https://github.com/peasant-labs/zone"
       description: "Sandboxed Docker workspaces for LLM coding agents"
       license: "MIT"

   release:
     github:
       owner: peasant-labs
       name: zone
   ```

   Key decisions:
   - `version: 2` header (required for GoReleaser v2)
   - `homebrew_casks` NOT `brews` (brews deprecated in GoReleaser v2.10)
   - `CGO_ENABLED=0` for cross-compilation from Linux CI runners
   - `windows/arm64` excluded (5 targets: linux/amd64, linux/arm64, darwin/amd64, darwin/arm64, windows/amd64)
   - `format_overrides` uses `formats: [zip]` for windows (tar.gz default for linux/darwin)
   - ldflags inject into `main.version`, `main.commit`, `main.date` matching the vars in main.go

2. Create `.github/workflows/ci.yml` with these exact contents:

   ```yaml
   name: CI

   on:
     push:
       branches: ["**"]
     pull_request:
       branches: ["**"]

   jobs:
     build-test:
       runs-on: ubuntu-latest
       steps:
         - uses: actions/checkout@v4
           with:
             fetch-depth: 0
         - uses: actions/setup-go@v5
           with:
             go-version: "1.24"
             cache: true
         - run: go build ./...
         - run: go test ./...

     lint:
       runs-on: ubuntu-latest
       steps:
         - uses: actions/checkout@v4
         - uses: actions/setup-go@v5
           with:
             go-version: "1.24"
             cache: true
         - uses: golangci/golangci-lint-action@v7
           with:
             version: v2.11

     govulncheck:
       runs-on: ubuntu-latest
       steps:
         - uses: actions/checkout@v4
         - uses: actions/setup-go@v5
           with:
             go-version: "1.24"
             cache: true
         - run: go install golang.org/x/vuln/cmd/govulncheck@latest
         - run: govulncheck ./...

     goreleaser-check:
       runs-on: ubuntu-latest
       steps:
         - uses: actions/checkout@v4
           with:
             fetch-depth: 0
         - uses: actions/setup-go@v5
           with:
             go-version: "1.24"
             cache: true
         - uses: goreleaser/goreleaser-action@v7
           with:
             version: "~> v2"
             args: release --snapshot --clean
           env:
             GORELEASER_CURRENT_TAG: v0.0.0-dev
   ```

   Key decisions:
   - 4 parallel jobs: `build-test`, `lint`, `govulncheck`, `goreleaser-check` (all independent)
   - `fetch-depth: 0` on checkout for build-test and goreleaser-check (GoReleaser needs full history)
   - `go-version: "1.24"` on all jobs (locked in CONTEXT.md)
   - `cache: true` on setup-go for faster builds
   - `golangci/golangci-lint-action@v7` with `version: v2.11` (not the full v2.11.4 — action handles patch)
   - `goreleaser release --snapshot --clean` (NOT just `goreleaser check` — snapshot actually cross-compiles, per CONTEXT.md)
   - `GORELEASER_CURRENT_TAG: v0.0.0-dev` env var on goreleaser step (prevents "no tags found" error on fresh repo)
   - Linux-only runners (ubuntu-latest) per CONTEXT.md — no macOS CI runners
   - Triggers on push to ANY branch and PRs to ANY branch

3. Create `.github/workflows/release.yml` with these exact contents:

   ```yaml
   name: Release

   on:
     push:
       tags:
         - "v*"

   permissions:
     contents: write

   jobs:
     release:
       runs-on: ubuntu-latest
       steps:
         - uses: actions/checkout@v4
           with:
             fetch-depth: 0
         - uses: actions/setup-go@v5
           with:
             go-version: "1.24"
             cache: true
         - uses: goreleaser/goreleaser-action@v7
           with:
             version: "~> v2"
             args: release --clean
           env:
             GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
             HOMEBREW_TAP_TOKEN: ${{ secrets.HOMEBREW_TAP_TOKEN }}
   ```

   Key decisions:
   - Triggers on `v*` tag pushes only (e.g., `v1.0.0`, `v0.1.0`)
   - `permissions: contents: write` for creating GitHub releases
   - `GITHUB_TOKEN` for GitHub release creation
   - `HOMEBREW_TAP_TOKEN` secret for writing to `peasant-labs/homebrew-tap` repo (user must configure this secret in GitHub repo settings)
   - `fetch-depth: 0` required for GoReleaser changelog generation
   - `--clean` flag removes previous dist/ artifacts
