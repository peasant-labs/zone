<objective>
Fix hostname glob enforcement so that glob patterns work end-to-end in both whitelist and blocklist modes, with clear warnings for DNS-unresolvable patterns and proper storage for refresh-time evaluation.

Purpose: The current BuildRuleSet hard-errors on whitelist allow globs and silently skips blocklist deny globs, making glob-based network rules non-functional. This blocks NET-12 (hostname glob matching for network rules). The fix stores glob patterns in the RuleSet for matcher-based evaluation and downgrades the hard error to a warning.

Output: BuildRuleSet stores globs in RuleSet.DenyGlobs/AllowGlobs fields, emits warnings for unresolvable globs, uses MatchAny for deny-before-allow glob evaluation, and the refresh loop evaluates newly-resolved hostnames against stored glob patterns. Tests cover all paths.

func RulesEqual(a, b RuleSet) bool {
    if a.Mode != b.Mode {
        return false
    }
    if len(a.AllowedIPs) != len(b.AllowedIPs) || len(a.DeniedIPs) != len(b.DeniedIPs) {
        return false
    }
    if len(a.DenyGlobs) != len(b.DenyGlobs) || len(a.AllowGlobs) != len(b.AllowGlobs) {
        return false
    }
    for ip := range a.AllowedIPs {
        if _, ok := b.AllowedIPs[ip]; !ok {
            return false
        }
    }
    for ip := range a.DeniedIPs {
        if _, ok := b.DeniedIPs[ip]; !ok {
            return false
        }
    }
    for i, g := range a.DenyGlobs {
        if g.String() != b.DenyGlobs[i].String() {
            return false
        }
    }
    for i, g := range a.AllowGlobs {
        if g.String() != b.AllowGlobs[i].String() {
            return false
        }
    }
    return true
}

func (f *Firewall) refreshOnce(ctx context.Context, cfg *config.NetworkConfig) error {
    // Suppress warnings during periodic refresh (already shown at launch time)
    old := warnWriter
    warnWriter = io.Discard
    newRules, err := BuildRuleSet(cfg, f.resolveFn)
    warnWriter = old
    if err != nil {
        return err
    }

    f.mu.Lock()
    defer f.mu.Unlock()

    if RulesEqual(f.rules, newRules) {
        return nil
    }
    if f.rules.Mode != "none" {
        if err := removeRulesForHash(ctx, f.execFn, f.containerHash); err != nil {
            return err
        }
    }
    return f.applyUnlocked(ctx, newRules)
}
