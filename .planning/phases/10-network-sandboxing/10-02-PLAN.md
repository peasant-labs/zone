```go
package network

import (
    "fmt"
    "net"
    "strings"

    "github.com/peasant-labs/zone/internal/config"
)

// RuleSet holds the resolved iptables rule parameters for a single container.
type RuleSet struct {
    Mode       string            // "whitelist", "blocklist", or "none"
    AllowedIPs map[string]string // IP -> hostname (for comments in rules)
    DeniedIPs  map[string]string // IP -> hostname
}

// BuildRuleSet generates a RuleSet from a merged NetworkConfig.
// It resolves all allow/deny hostnames to IPs on the host.
// The denyPatterns are compiled from config.Deny. In whitelist mode,
// hostnames matching any deny pattern are excluded from the allow set (D-09).
//
// IMPORTANT: Glob patterns (e.g., "*.anthropic.com") in the whitelist allow list
// are rejected with an error. Glob patterns cannot be resolved to IP addresses
// for iptables rules. Only literal hostnames are supported in whitelist allow lists.
// Glob patterns ARE supported in deny lists (for deny-first matching against literals).
//
// The resolveFunc parameter allows injection of a custom resolver for testing.
// Pass nil to use net.LookupHost.
func BuildRuleSet(cfg *config.NetworkConfig, resolveFunc func(string) ([]string, error)) (RuleSet, error) {
    mode := normalizeMode(cfg.Mode)
    if mode == "none" {
        return RuleSet{Mode: "none"}, nil
    }

    if resolveFunc == nil {
        resolveFunc = net.LookupHost
    }

    rs := RuleSet{
        Mode:       mode,
        AllowedIPs: make(map[string]string),
        DeniedIPs:  make(map[string]string),
    }

    // Compile deny patterns for deny-first evaluation
    denyPatterns, err := CompileAll(cfg.Deny)
    if err != nil {
        return rs, fmt.Errorf("compile deny patterns: %w", err)
    }

    if mode == "whitelist" {
        // Compile and resolve allow patterns
        for _, pattern := range cfg.Allow {
            cp, err := Compile(pattern)
            if err != nil {
                return rs, fmt.Errorf("compile allow pattern %q: %w", pattern, err)
            }
            if cp.IsGlob() {
                // Glob patterns cannot be resolved to IP addresses for iptables.
                // Return a clear error rather than silently skipping (NET-01 + NET-12).
                return rs, fmt.Errorf("glob patterns in whitelist allow list are not supported in Phase 1: %q cannot be resolved to IP addresses for iptables rules. Use exact hostnames instead (e.g., \"api.anthropic.com\" instead of \"*.anthropic.com\")", pattern)
            }
            // Check deny-first: if this literal hostname matches any deny pattern, skip it
            if MatchAny(cp.String(), denyPatterns) {
                continue
            }
            ips, err := resolveFunc(cp.String())
            if err != nil {
                // Resolution failure: skip this hostname, don't fail the whole set
                continue
            }
            for _, ip := range ips {
                rs.AllowedIPs[ip] = cp.String()
            }
        }
    }

    if mode == "blocklist" || mode == "whitelist" {
        // Resolve deny hostnames to IPs
        for _, pattern := range cfg.Deny {
            cp, err := Compile(pattern)
            if err != nil {
                return rs, fmt.Errorf("compile deny pattern %q: %w", pattern, err)
            }
            if cp.IsGlob() {
                continue // Globs in deny list are used for pattern matching, not IP resolution
            }
            ips, err := resolveFunc(cp.String())
            if err != nil {
                continue
            }
            for _, ip := range ips {
                rs.DeniedIPs[ip] = cp.String()
            }
        }
    }

    return rs, nil
}

// normalizeMode converts the config mode string to a canonical form.
// Empty string defaults to "none" per D-15/D-16.
func normalizeMode(mode string) string {
    m := strings.ToLower(strings.TrimSpace(mode))
    switch m {
    case "whitelist", "blocklist":
        return m
    default:
        return "none"
    }
}

// RulesEqual returns true if two RuleSets have identical IP sets.
func RulesEqual(a, b RuleSet) bool {
    if a.Mode != b.Mode {
        return false
    }
    if len(a.AllowedIPs) != len(b.AllowedIPs) || len(a.DeniedIPs) != len(b.DeniedIPs) {
        return false
    }
    for ip := range a.AllowedIPs {
        if _, ok := b.AllowedIPs[ip]; !ok {
            return false
        }
    }
    for ip := range a.DeniedIPs {
        if _, ok := b.DeniedIPs[ip]; !ok {
            return false
        }
    }
    return true
}
```

```go
package network

import (
    "context"
    "fmt"
    "os"
    "os/exec"
    "path/filepath"
    "regexp"
    "strings"
    "sync"
    "time"

    "github.com/peasant-labs/zone/internal/config"
)

// ExecFunc is the function signature for running iptables commands.
// Default implementation calls sudo iptables. Tests inject a mock.
type ExecFunc func(ctx context.Context, args ...string) ([]byte, error)

// DefaultExecFunc runs: sudo iptables [args...]
func DefaultExecFunc(ctx context.Context, args ...string) ([]byte, error) {
    fullArgs := append([]string{"iptables"}, args...)
    cmd := exec.CommandContext(ctx, "sudo", fullArgs...)
    return cmd.CombinedOutput()
}

// Firewall manages iptables rules for a single container.
type Firewall struct {
    containerHash string   // e.g. "abc123def456" (16 chars from container name hash)
    bridgeIface   string   // Docker bridge interface name, e.g. "br-abc123def456"
    cacheDir      string   // path to .zone/ directory for firewall.rules
    rules         RuleSet  // last-applied rules
    mu            sync.Mutex
    execFn        ExecFunc // injectable for testing
}

// NewFirewall creates a Firewall for a specific container.
// containerHash is the 16-char hex from the container name (e.g. from ContainerName).
// bridgeIface is the Docker bridge interface name.
// cacheDir is the path to .zone/ for writing firewall.rules.
func NewFirewall(containerHash, bridgeIface, cacheDir string, execFn ExecFunc) *Firewall {
    if execFn == nil {
        execFn = DefaultExecFunc
    }
    return &Firewall{
        containerHash: containerHash,
        bridgeIface:   bridgeIface,
        cacheDir:      cacheDir,
        execFn:        execFn,
    }
}

// tag returns the iptables comment tag for this container.
func (f *Firewall) tag() string {
    return "zone-" + f.containerHash
}

// defaultTag returns the tag for the DROP default rule.
func (f *Firewall) defaultTag() string {
    return f.tag() + "-default"
}

// Apply sets up iptables rules based on the RuleSet.
// In whitelist mode: INSERT ACCEPT rules at position 1 (for allowed IPs + DNS), then APPEND DROP default.
// In blocklist mode: INSERT DROP rules at position 1 for denied IPs.
// In mode=none: no-op.
// Writes firewall.rules cache file on success.
func (f *Firewall) Apply(ctx context.Context, rs RuleSet) error {
    f.mu.Lock()
    defer f.mu.Unlock()

    if rs.Mode == "none" {
        f.rules = rs
        return nil
    }

    tag := f.tag()
    var appliedRules []string

    if rs.Mode == "whitelist" {
        // 1. Allow Docker embedded DNS (127.0.0.11:53) -- must come first (D-11)
        for _, proto := range []string{"udp", "tcp"} {
            args := []string{"-I", "FORWARD", "1",
                "-i", f.bridgeIface,
                "-d", "127.0.0.11",
                "-p", proto, "--dport", "53",
                "-j", "ACCEPT",
                "-m", "comment", "--comment", tag}
            if _, err := f.execFn(ctx, args...); err != nil {
                return fmt.Errorf("allow DNS %s: %w", proto, err)
            }
            appliedRules = append(appliedRules, "sudo iptables "+strings.Join(args, " "))
        }

        // 2. ACCEPT for each allowed IP (D-10)
        for ip, hostname := range rs.AllowedIPs {
            args := []string{"-I", "FORWARD", "1",
                "-i", f.bridgeIface,
                "-d", ip,
                "-j", "ACCEPT",
                "-m", "comment", "--comment", tag}
            if _, err := f.execFn(ctx, args...); err != nil {
                return fmt.Errorf("allow IP %s (%s): %w", ip, hostname, err)
            }
            appliedRules = append(appliedRules, fmt.Sprintf("sudo iptables %s  # %s", strings.Join(args, " "), hostname))
        }

        // 3. DROP for each denied IP (deny wins -- D-08, D-09)
        for ip, hostname := range rs.DeniedIPs {
            args := []string{"-I", "FORWARD", "1",
                "-i", f.bridgeIface,
                "-d", ip,
                "-j", "DROP",
                "-m", "comment", "--comment", tag}
            if _, err := f.execFn(ctx, args...); err != nil {
                return fmt.Errorf("deny IP %s (%s): %w", ip, hostname, err)
            }
            appliedRules = append(appliedRules, fmt.Sprintf("sudo iptables %s  # %s", strings.Join(args, " "), hostname))
        }

        // 4. DROP default: append (so it's evaluated last) -- D-08
        args := []string{"-A", "FORWARD",
            "-i", f.bridgeIface,
            "-j", "DROP",
            "-m", "comment", "--comment", f.defaultTag()}
        if _, err := f.execFn(ctx, args...); err != nil {
            return fmt.Errorf("drop default: %w", err)
        }
        appliedRules = append(appliedRules, "sudo iptables "+strings.Join(args, " "))
    }

    if rs.Mode == "blocklist" {
        // DROP for each denied IP -- D-13
        for ip, hostname := range rs.DeniedIPs {
            args := []string{"-I", "FORWARD", "1",
                "-i", f.bridgeIface,
                "-d", ip,
                "-j", "DROP",
                "-m", "comment", "--comment", tag}
            if _, err := f.execFn(ctx, args...); err != nil {
                return fmt.Errorf("deny IP %s (%s): %w", ip, hostname, err)
            }
            appliedRules = append(appliedRules, fmt.Sprintf("sudo iptables %s  # %s", strings.Join(args, " "), hostname))
        }
    }

    f.rules = rs

    // Write firewall.rules cache (D-31, D-32, D-33)
    if err := f.writeRulesCache(appliedRules); err != nil {
        // Non-fatal: rules are applied even if cache write fails
        fmt.Fprintf(os.Stderr, "Warning: failed to write firewall.rules cache: %v\n", err)
    }

    return nil
}

// Remove tears down all iptables rules tagged with this container's hash.
// It lists all rules with iptables -S, finds those matching the zone-{hash} comment,
// and deletes them by converting -A/-I to -D.
func (f *Firewall) Remove(ctx context.Context) error {
    f.mu.Lock()
    defer f.mu.Unlock()

    return removeRulesForHash(ctx, f.execFn, f.containerHash)
}

// removeRulesForHash removes all iptables rules tagged with a specific zone hash.
// Shared by both Firewall.Remove() and stale rule cleanup.
func removeRulesForHash(ctx context.Context, execFn ExecFunc, hash string) error {
    out, err := execFn(ctx, "-S")
    if err != nil {
        return fmt.Errorf("list iptables rules: %w", err)
    }

    tag := "zone-" + hash
    defaultTag := tag + "-default"

    for _, line := range strings.Split(string(out), "\n") {
        line = strings.TrimSpace(line)
        if !strings.Contains(line, tag) && !strings.Contains(line, defaultTag) {
            continue
        }
        // Convert -A FORWARD ... to -D FORWARD ...
        // Convert -I FORWARD ... to -D FORWARD ...
        delRule := line
        if strings.HasPrefix(delRule, "-A ") {
            delRule = "-D " + delRule[3:]
        } else if strings.HasPrefix(delRule, "-I ") {
            delRule = "-D " + delRule[3:]
        } else {
            continue // skip non-add/insert lines
        }
        args := strings.Fields(delRule)
        if _, err := execFn(ctx, args...); err != nil {
            // Best-effort removal: log but continue
            fmt.Fprintf(os.Stderr, "Warning: failed to remove rule: %s: %v\n", delRule, err)
        }
    }

    return nil
}

// zoneHashRe matches zone comment tags in iptables -S output.
var zoneHashRe = regexp.MustCompile(`zone-([a-f0-9]+?)(?:-default)?(?:\s|$|")`)

// CleanStaleRules removes iptables rules for zone containers that are no longer running.
// runningHashes is the set of container hashes that are currently active.
func CleanStaleRules(ctx context.Context, execFn ExecFunc, runningHashes map[string]bool) error {
    if execFn == nil {
        execFn = DefaultExecFunc
    }
    out, err := execFn(ctx, "-S")
    if err != nil {
        return fmt.Errorf("list iptables rules for stale detection: %w", err)
    }

    // Extract all unique zone hashes from iptables output
    foundHashes := make(map[string]bool)
    for _, match := range zoneHashRe.FindAllSubmatch(out, -1) {
        foundHashes[string(match[1])] = true
    }

    // Remove rules for hashes not in the running set
    for hash := range foundHashes {
        if !runningHashes[hash] {
            if err := removeRulesForHash(ctx, execFn, hash); err != nil {
                fmt.Fprintf(os.Stderr, "Warning: failed to clean stale rules for zone-%s: %v\n", hash, err)
            }
        }
    }

    return nil
}

// StartRefresh starts a background goroutine that re-resolves hostnames every 5 minutes
// and updates iptables rules if IPs have changed.
// Cancel the context to stop the goroutine.
func (f *Firewall) StartRefresh(ctx context.Context, cfg *config.NetworkConfig) {
    go func() {
        ticker := time.NewTicker(5 * time.Minute)
        defer ticker.Stop()
        for {
            select {
            case <-ctx.Done():
                return
            case <-ticker.C:
                newRS, err := BuildRuleSet(cfg, nil)
                if err != nil {
                    continue
                }
                f.mu.Lock()
                if !RulesEqual(f.rules, newRS) {
                    // Remove old rules, apply new ones
                    _ = removeRulesForHash(ctx, f.execFn, f.containerHash)
                    _ = f.applyUnlocked(ctx, newRS)
                }
                f.mu.Unlock()
            }
        }
    }()
}

// applyUnlocked is Apply without the mutex -- called from StartRefresh which already holds it.
func (f *Firewall) applyUnlocked(ctx context.Context, rs RuleSet) error {
    // Same logic as Apply but without lock
    tag := f.tag()
    var appliedRules []string

    if rs.Mode == "whitelist" {
        for _, proto := range []string{"udp", "tcp"} {
            args := []string{"-I", "FORWARD", "1", "-i", f.bridgeIface, "-d", "127.0.0.11", "-p", proto, "--dport", "53", "-j", "ACCEPT", "-m", "comment", "--comment", tag}
            f.execFn(ctx, args...)
            appliedRules = append(appliedRules, "sudo iptables "+strings.Join(args, " "))
        }
        for ip, hostname := range rs.AllowedIPs {
            args := []string{"-I", "FORWARD", "1", "-i", f.bridgeIface, "-d", ip, "-j", "ACCEPT", "-m", "comment", "--comment", tag}
            f.execFn(ctx, args...)
            appliedRules = append(appliedRules, fmt.Sprintf("sudo iptables %s  # %s", strings.Join(args, " "), hostname))
        }
        for ip, hostname := range rs.DeniedIPs {
            args := []string{"-I", "FORWARD", "1", "-i", f.bridgeIface, "-d", ip, "-j", "DROP", "-m", "comment", "--comment", tag}
            f.execFn(ctx, args...)
            appliedRules = append(appliedRules, fmt.Sprintf("sudo iptables %s  # %s", strings.Join(args, " "), hostname))
        }
        args := []string{"-A", "FORWARD", "-i", f.bridgeIface, "-j", "DROP", "-m", "comment", "--comment", f.defaultTag()}
        f.execFn(ctx, args...)
        appliedRules = append(appliedRules, "sudo iptables "+strings.Join(args, " "))
    }

    if rs.Mode == "blocklist" {
        for ip, hostname := range rs.DeniedIPs {
            args := []string{"-I", "FORWARD", "1", "-i", f.bridgeIface, "-d", ip, "-j", "DROP", "-m", "comment", "--comment", tag}
            f.execFn(ctx, args...)
            appliedRules = append(appliedRules, fmt.Sprintf("sudo iptables %s  # %s", strings.Join(args, " "), hostname))
        }
    }

    f.rules = rs
    _ = f.writeRulesCache(appliedRules)
    return nil
}

// writeRulesCache writes human-readable iptables rules to .zone/firewall.rules (D-31 through D-33).
func (f *Firewall) writeRulesCache(rules []string) error {
    if f.cacheDir == "" {
        return nil
    }
    content := "# zone firewall rules for zone-" + f.containerHash + "\n"
    content += "# Generated: " + time.Now().Format(time.RFC3339) + "\n"
    content += "# Mode: " + f.rules.Mode + "\n\n"
    content += strings.Join(rules, "\n") + "\n"
    path := filepath.Join(f.cacheDir, "firewall.rules")
    return os.WriteFile(path, []byte(content), 0644)
}

// RemoveRulesCache deletes the .zone/firewall.rules file.
func (f *Firewall) RemoveRulesCache() error {
    if f.cacheDir == "" {
        return nil
    }
    path := filepath.Join(f.cacheDir, "firewall.rules")
    if err := os.Remove(path); err != nil && !os.IsNotExist(err) {
        return err
    }
    return nil
}
```
