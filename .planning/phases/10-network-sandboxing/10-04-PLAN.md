<objective>
Fix durable firewall cleanup so that zone stop, zone destroy, and zone clean all remove iptables rules even when run in a fresh process that never called Launch.

Purpose: The current implementation only removes firewall rules through the in-memory `m.firewall` pointer, which is nil in fresh CLI processes. This means running `zone stop` from a different terminal or after restarting the shell leaves orphaned iptables rules on the host. This is a blocker for NET-05 (tagged cleanup).

Output: Manager.Stop reconstructs firewall state from cache/naming when m.firewall is nil. cmd/clean.go invokes firewall cleanup before deleting .zone/. Regression tests verify both paths.
</objective>

**3. Add reconstructFirewallForCleanup method to Manager in internal/docker/manager.go:**

Add this method after the Stop method (around line 394):

```go
// reconstructFirewallForCleanup creates a Firewall instance from cached/derived
// state for cleanup purposes. Used when m.firewall is nil (fresh process that
// didn't launch the container) but firewall rules may exist on the host.
//
// Returns nil if:
// - Network mode is "none" or empty (no firewall rules to clean)
// - Platform doesn't support iptables
// - repoDir is empty (can't derive container hash)
func (m *Manager) reconstructFirewallForCleanup(ctx context.Context) *network.Firewall {
    mode := m.config.Network.Mode
    if mode == "" || strings.ToLower(mode) == "none" {
        return nil
    }
    if !m.platform.SupportsIPTables {
        return nil
    }

    containerName := ContainerName(m.repoDir)
    if len(containerName) < 16 {
        return nil
    }
    containerHash := containerName[len(containerName)-16:]

    netID, err := m.cache.NetworkID()
    if err != nil || netID == "" {
        // No cached network ID -- use empty bridge iface.
        // RemoveRulesByHash works by tag matching, not bridge iface,
        // so this is fine for cleanup.
    }
    bridgeIface := ""
    if netID != "" {
        bridgeIface = m.BridgeInterfaceName(ctx, netID)
    }

    return network.NewFirewall(containerHash, bridgeIface, m.cache.Dir(), nil)
}
```

Replace the RunE function body with:

```go
RunE: func(cmd *cobra.Command, args []string) error {
    cwd, err := os.Getwd()
    if err != nil {
        return fmt.Errorf("get working directory: %w", err)
    }
    c := cache.New(cwd)

    // Check if lock is held; warn but proceed (per CONTEXT.md decision)
    if pid := cache.ReadLockPID(c.Dir()); pid > 0 {
        fmt.Fprintf(os.Stderr,
            "Warning: another zone process (PID %d) may be running. Cleaning anyway.\n", pid)
    }

    // Attempt firewall cleanup before deleting cache (D-39).
    // Uses the container hash derived from the repo path and cleans
    // tagged iptables rules. Best-effort: if sudo/iptables unavailable
    // or not on Linux, this is silently skipped.
    if runtime.GOOS == "linux" {
        containerName := docker.ContainerName(cwd)
        if len(containerName) >= 16 {
            containerHash := containerName[len(containerName)-16:]
            ctx, cancel := signal.NotifyContext(context.Background(), os.Interrupt, syscall.SIGTERM)
            cleanErr := network.RemoveRulesByHash(ctx, nil, containerHash)
            cancel()
            if cleanErr != nil {
                fmt.Fprintf(os.Stderr, "Warning: could not remove firewall rules: %v\n", cleanErr)
            }
        }
    }

    if err := c.Clean(); err != nil {
        return fmt.Errorf("clean cache: %w", err)
    }

    fmt.Fprintln(cmd.OutOrStdout(), "Removed .zone/ cache directory.")

    removeImage, _ := cmd.Flags().GetBool("image")
    if removeImage {
        ctx, cancel := signal.NotifyContext(context.Background(), os.Interrupt, syscall.SIGTERM)
        defer cancel()

        cfg, _, cfgErr := config.LoadMerged(cwd)
        if cfgErr != nil {
            return fmt.Errorf("load config for --image: %w", cfgErr)
        }
        // Re-create a fresh cache (dir was just cleaned).
        freshCache := cache.New(cwd)
        mgr, mgrErr := docker.NewManager(cfg, freshCache, cwd, version)
        if mgrErr != nil {
            return mgrErr
        }
        if err2 := mgr.RemoveImage(ctx); err2 != nil {
            return fmt.Errorf("remove image: %w", err2)
        }
        fmt.Fprintln(cmd.OutOrStdout(), "Removed Docker image.")
    }

    return nil
},
```
