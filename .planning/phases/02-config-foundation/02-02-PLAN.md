   // mergeUnion: deduplicated union (global first, repo additions after)
   func mergeUnion(global, repo []string) ([]string, []AnnotatedListItem) {
       seen := make(map[string]bool)
       var result []string
       var annotated []AnnotatedListItem
       for _, v := range global {
           if !seen[v] {
               seen[v] = true
               result = append(result, v)
               annotated = append(annotated, AnnotatedListItem{Value: v, Source: SourceGlobal})
           }
       }
       for _, v := range repo {
           if !seen[v] {
               seen[v] = true
               result = append(result, v)
               annotated = append(annotated, AnnotatedListItem{Value: v, Source: SourceRepo})
           }
       }
       return result, annotated
   }

   // mergeAppend: global + repo concatenated (no dedup)
   func mergeAppend(global, repo []string) ([]string, []AnnotatedListItem) {
       var result []string
       var annotated []AnnotatedListItem
       for _, v := range global {
           result = append(result, v)
           annotated = append(annotated, AnnotatedListItem{Value: v, Source: SourceGlobal})
       }
       for _, v := range repo {
           result = append(result, v)
           annotated = append(annotated, AnnotatedListItem{Value: v, Source: SourceRepo})
       }
       return result, annotated
   }

   // mergeReplace: repo replaces global entirely if repo is non-empty
   func mergeReplace(global, repo []string) ([]string, []AnnotatedListItem) {
       source := SourceGlobal
       vals := global
       if len(repo) > 0 {
           source = SourceRepo
           vals = repo
       }
       var annotated []AnnotatedListItem
       for _, v := range vals {
           annotated = append(annotated, AnnotatedListItem{Value: v, Source: source})
       }
       return vals, annotated
   }

   ```go
   func buildSymlinkChain(path string) string {
       var chain []string
       chain = append(chain, path)
       current := path
       for i := 0; i < 10; i++ { // prevent infinite loops
           target, err := os.Readlink(current)
           if err != nil {
               break // not a symlink or error
           }
           if !filepath.IsAbs(target) {
               target = filepath.Join(filepath.Dir(current), target)
           }
           chain = append(chain, target)
           current = target
       }
       if len(chain) <= 1 {
           return ""
       }
       return strings.Join(chain, " -> ")
   }
   ```

   func NormalizeMountPermission(mountSpec string) (string, error) {
       parts := strings.Split(mountSpec, ":")
       switch len(parts) {
       case 2: // "/host:/container" — no permission specified, default to ro
           return mountSpec + ":ro", nil
       case 3: // "/host:/container:ro|rw"
           perm := parts[2]
           if perm != "ro" && perm != "rw" {
               return "", fmt.Errorf("invalid mount permission %q in %q (must be 'ro' or 'rw')", perm, mountSpec)
           }
           return mountSpec, nil
       default:
           return "", fmt.Errorf("invalid mount format %q (expected /host:/container[:ro|rw])", mountSpec)
       }
   }

   func Validate(cfg *MergedConfig) ValidationErrors {
       var errs ValidationErrors
       // 1. Mount validation: check each extra_mount
       for _, mount := range cfg.Workspace.ExtraMounts {
           // Normalize mount permission
           normalized, err := NormalizeMountPermission(mount)
           if err != nil {
               errs = append(errs, ValidationError{Category: "type_error", Message: err.Error(), Field: "workspace.extra_mounts"})
               continue
           }
           // Extract host path (first component)
           hostPath := strings.Split(normalized, ":")[0]
           // Resolve symlinks
           resolved, resolveErr := filepath.EvalSymlinks(hostPath)
           if resolveErr != nil && !os.IsNotExist(resolveErr) {
               errs = append(errs, ValidationError{Category: "warning", Message: fmt.Sprintf("cannot resolve mount path %s: %v", hostPath, resolveErr), Field: "workspace.extra_mounts"})
               continue
           }
           if resolveErr != nil && os.IsNotExist(resolveErr) {
               resolved = hostPath // path doesn't exist yet; check raw path
           }
           chain := buildSymlinkChain(hostPath)
           if reason, blocked := isBlockedMount(resolved); blocked {
               errs = append(errs, ValidationError{
                   Category: "dangerous_mount",
                   Message: (&DangerousMountError{Path: hostPath, Resolved: resolved, Chain: chain, Reason: reason}).Error(),
                   Field: "workspace.extra_mounts",
               })
           }
       }
