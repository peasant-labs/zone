**1. Create `cmd/errors.go`** with the mapError function:

```go
package cmd

import (
    "errors"
    "fmt"

    "github.com/peasant-labs/zone/internal/cache"
    "github.com/peasant-labs/zone/internal/config"
    "github.com/peasant-labs/zone/internal/docker"
)

// mapError maps a sentinel error to a user-facing remediation message and exit code.
// Exit codes per spec section 3.3:
//   0 = success
//   1 = generic/unknown
//   2 = config error
//   3 = Docker error
//   4 = network error
//   5 = cache/lock error
//   6 = no container
func mapError(err error) (string, int) {
    if err == nil {
        return "", 0
    }

    switch {
    // Exit code 6: no container
    case errors.Is(err, docker.ErrNoContainer):
        return "Error: No running zone container for this repo.\n\n" +
            "  Run `zone launch` to start one, or `zone ls` to see all containers.", 6

    // Exit code 5: cache/lock
    case errors.Is(err, cache.ErrLockContention):
        return "Error: Another zone process is operating on this repo.\n\n" +
            "  Wait for the other process to finish, or run `zone clean` if it crashed.", 5

    // Exit code 4: network
    case errors.Is(err, docker.ErrNetworkUnsupported):
        return "Error: Network sandboxing is not supported on this platform.\n\n" +
            "  Network filtering requires Linux with iptables.\n" +
            "  Set [network] mode = \"none\" to disable filtering.", 4

    // Exit code 3: Docker
    case errors.Is(err, docker.ErrDockerNotRunning):
        return "Error: Docker daemon is not running.\n\n" +
            "  macOS:  Open Docker Desktop, or run `open -a Docker`\n" +
            "  Linux:  Run `sudo systemctl start docker`\n\n" +
            "Zone requires Docker to create sandboxed workspaces.", 3

    // Exit code 2: config errors
    case errors.Is(err, config.ErrNoConfig):
        return "Error: No zone.toml found.\n\n" +
            "  Run `zone init --harness <name>` to create one,\n" +
            "  or `zone launch --harness <name>` for zero-config quickstart.", 2

    case errors.Is(err, config.ErrVersionMismatch):
        return "Error: Unsupported config version in zone.toml.\n\n" +
            "  Zone currently supports `version = 1`.\n" +
            "  Check zone.toml and update the version field.", 2

    default:
        // Check for UnknownKeysError (config category)
        var uke *config.UnknownKeysError
        if errors.As(err, &uke) {
            return "Error: " + err.Error(), 2
        }

        return "Error: " + err.Error(), 1
    }
}

// printRemediation writes a remediation message to stderr and returns the exit code.
// Used by wrapRunE to provide user-facing error messages.
func printRemediation(cmd_name string, err error) int {
    msg, code := mapError(err)
    // msg is printed by main.go after Execute() returns
    _ = msg
    _ = cmd_name
    return code
}
```

**2. Update `main.go`** with the full exit code taxonomy:

Replace the entire main.go with:

```go
package main

import (
    "errors"
    "fmt"
    "os"

    "github.com/peasant-labs/zone/cmd"
    "github.com/peasant-labs/zone/internal/cache"
    "github.com/peasant-labs/zone/internal/config"
    "github.com/peasant-labs/zone/internal/docker"
)

var version = "dev"
var commit = "none"
var date = "unknown"

func main() {
    cmd.SetVersion(version, commit, date)
    if err := cmd.Execute(); err != nil {
        // Map error to exit code using the spec section 3.3 taxonomy.
        // Order matters: specific errors checked before generic fallback.
        // D-24: errors.Is() chain — specific before generic.

        var exitCode int

        switch {
        case errors.Is(err, docker.ErrNoContainer):
            exitCode = 6
        case errors.Is(err, cache.ErrLockContention):
            exitCode = 5
        case errors.Is(err, docker.ErrNetworkUnsupported):
            exitCode = 4
        case errors.Is(err, docker.ErrDockerNotRunning):
            exitCode = 3
        case errors.Is(err, config.ErrNoConfig),
             errors.Is(err, config.ErrVersionMismatch):
            exitCode = 2
        default:
            var uke *config.UnknownKeysError
            if errors.As(err, &uke) {
                exitCode = 2
            } else {
                exitCode = 1
            }
        }

        // Remediation hints are printed to stderr by the RunE wrappers.
        // main.go only handles exit code mapping.
        // If SilenceErrors is true, Cobra won't print the error, so we print it here.
        fmt.Fprintf(os.Stderr, "%s\n", err)

        os.Exit(exitCode)
    }
}
```

**Concrete example for cmd/stop.go** (apply same pattern to all):

```go
RunE: func(cmd *cobra.Command, args []string) error {
    ctx, cancel := signal.NotifyContext(context.Background(), os.Interrupt, syscall.SIGTERM)
    defer cancel()

    cwd, err := os.Getwd()
    if err != nil {
        return fmt.Errorf("get working directory: %w", err)
    }

    cfg, _, err := config.LoadMerged(cwd)
    if err != nil {
        return fmt.Errorf("load config: %w", err)
    }

    c := cache.New(cwd)
    mgr, err := docker.NewManager(cfg, c, cwd, version)
    if err != nil {
        return err
    }

    return mgr.Stop(ctx)
},
```

**D-28 note for launch specifically:** The `docker exec -it` subprocess in `attachInteractive` is spawned via `os/exec.Command`. When Ctrl+C is pressed, SIGINT goes to the entire process group (including the subprocess), which naturally sends SIGINT to the harness. The `signal.NotifyContext` cancels the Go context but does NOT kill the subprocess directly — the `exec.Command.Run()` returns when the subprocess exits. Container stays alive per spec.
