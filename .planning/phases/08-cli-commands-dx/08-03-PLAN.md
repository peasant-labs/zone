// TestAliases verifies all command aliases resolve correctly (DX-08, D-37).
func TestAliases(t *testing.T) {
    binary := getZoneBinary(t)

    aliases := []struct {
        alias   string
        primary string
    }{
        {"up", "launch"},
        {"down", "stop"},
        {"list", "ls"},
        {"log", "logs"},
        {"st", "status"},
    }

    for _, a := range aliases {
        t.Run(a.alias, func(t *testing.T) {
            // Verify alias --help matches primary --help Short description
            aliasOut, err := exec.Command(binary, a.alias, "--help").CombinedOutput()
            require.NoError(t, err, "alias %s --help failed: %s", a.alias, string(aliasOut))

            primaryOut, err := exec.Command(binary, a.primary, "--help").CombinedOutput()
            require.NoError(t, err, "primary %s --help failed: %s", a.primary, string(primaryOut))

            // Both should have the same Short description line
            // The help output starts with "Usage:" line, then descriptions
            assert.Contains(t, string(aliasOut), string(extractShortLine(primaryOut)),
                "alias %s help does not match primary %s", a.alias, a.primary)
        })
    }
}

// extractShortLine extracts the first non-empty line after trimming usage info.
func extractShortLine(help []byte) string {
    lines := strings.Split(string(help), "\n")
    for _, l := range lines {
        l = strings.TrimSpace(l)
        if l != "" && !strings.HasPrefix(l, "Usage:") && !strings.HasPrefix(l, "Aliases:") {
            return l
        }
    }
    return ""
}

// TestExitCode6OnNoContainer verifies no-container errors produce exit code 6 (DX-01).
func TestExitCode6OnNoContainer(t *testing.T) {
    binary := getZoneBinary(t)
    dir := t.TempDir()

    // Write valid zone.toml so config loads, but no container exists
    os.WriteFile(filepath.Join(dir, "zone.toml"),
        []byte("version = 1\nharness = \"claude-code\"\n"), 0644)
    // Create .zone/ with empty container_id
    zoneDir := filepath.Join(dir, ".zone")
    os.MkdirAll(zoneDir, 0755)

    cmd := exec.Command(binary, "status")
    cmd.Dir = dir
    cmd.Env = append(os.Environ(), "XDG_CONFIG_HOME="+dir+"/no-xdg")
    _ = cmd.Run()

    // This may produce exit 3 (Docker not running) or 6 (no container).
    // Both are valid since they depend on Docker daemon state.
    code := cmd.ProcessState.ExitCode()
    assert.True(t, code == 3 || code == 6,
        "expected exit code 3 or 6 for status with no container, got %d", code)
}
